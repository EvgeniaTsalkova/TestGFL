CREATE TABLE equation (
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY ,
    equation varchar(150) NOT NULL UNIQUE
);

CREATE TABLE root (
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY ,
    value real NOT NULL UNIQUE
);

CREATE TABLE equation_root (
    equation_id int REFERENCES equation(id),
    root_id int REFERENCES root(id),
    PRIMARY KEY(equation_id, root_id)
);
